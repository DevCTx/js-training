function updateRecords(record, id, prop, value) {
  if (value===null || value===undefined || value==="") {
    delete record[id][prop];
  }
  else {  
    if (prop !== "tracks" ) {
      record[id][prop] = value;
    }
    else { // prop == "tracks"
      if (record[id]["tracks"] == undefined) {
        record[id]["tracks"]=[value];        
      }
      else {
        record[id]["tracks"].push(value);
      }
    }
  }
  return record;
}
